    NAME AS OWNER_NAME, 
    COUNT(REPORT_NUMBER) AS NUMBER_OF_ACCIDENTS, 
    SUM(DAMAGE_AMOUNT) AS TOTAL_DAMAGE 
FROM 
    PARTICIPATED 
    INNER JOIN PERSON ON PERSON.DRIVERID = PARTICIPATED.DRIVERID 
GROUP BY 
    NAME 
HAVING 
    COUNT(REPORT_NUMBER) > 0 
ORDER BY 
    TOTAL_DAMAGE DESC;
